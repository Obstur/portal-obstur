import streamlit as st
import pandas as pd
import plotly.express as px

# Configuração da página (Minimalista)
st.set_page_config(page_title="Painel de Turismo", layout="wide")

# Tema escuro para os gráficos
TEMPLATE = "plotly_dark"

# Links Públicos do Google Sheets
URL_EMPREGABILIDADE = "https://docs.google.com/spreadsheets/d/e/2PACX-1vQT-EtDg1pO7uGcQf8yHNiLKb2euWDVDOP84yaZKzOPQNlssQbhbR4kSHJ5HNqbVt1NS8VPFsFDm5dp/pub?output=csv"
URL_CADASTUR = "https://docs.google.com/spreadsheets/d/e/2PACX-1vScKBQfjVOP_WRtCo7QilgnPPGh3cA2VCT-RUnrJ_KoxxLixbTmcFAsfuG9EONT5F7bciZIBCn6Rwmw/pub?output=csv"
URL_FLUXO = "https://docs.google.com/spreadsheets/d/e/2PACX-1vQKgD0hlDacayEi-hCuWnZC37z-etoULtAG0DTrmnVbAMJc_A3Eo_EqapaE_3hCkA8Dudcv3kdOFbbT/pub?output=csv"

# Cache para otimizar leitura
@st.cache_data(ttl=3600)
def carregar_dados(url):
    try:
        return pd.read_csv(url, on_bad_lines='skip')
    except Exception as e:
        return pd.DataFrame()

# Carregando os dados
df_emp = carregar_dados(URL_EMPREGABILIDADE)
df_cad = carregar_dados(URL_CADASTUR)
df_fluxo = carregar_dados(URL_FLUXO)

# Título do Painel
st.title("📊 Observatório de Turismo")
st.markdown("---")

# Criando as 3 Abas
aba1, aba2, aba3 = st.tabs(["Empregabilidade", "Cadastur", "Fluxo Turístico"])

with aba1:
    st.subheader("Evolução da Empregabilidade")
    if not df_emp.empty:
        st.dataframe(df_emp.head()) # Substitua por gráficos Plotly após limpar formato
    else:
        st.warning("Aguardando formatação dos dados de Empregabilidade.")

with aba2:
    st.subheader("Registros Cadastur")
    if not df_cad.empty:
        st.dataframe(df_cad.head())
    else:
        st.warning("Aguardando formatação dos dados do Cadastur.")

with aba3:
    st.subheader("Fluxo Turístico nos Atrativos")
    if not df_fluxo.empty:
        st.dataframe(df_fluxo.head())
    else:
        st.warning("Aguardando formatação dos dados de Fluxo.")

# Estilo customizado via CSS
st.markdown("""
